   ```python
   @pytest.mark.parametrize("command", all_commands())
   def test_all_commands_json_output_format(command, runner):
       """验证所有命令的 JSON 输出格式有效"""
       with patch('gallery_dl_auo.api.pixiv_client.PixivClient'):
           result = runner.invoke(cli, ['--json-output', command])

           # stdout 应为有效 JSON
           try:
               output_data = json.loads(result.stdout)
           except json.JSONDecodeError:
               pytest.fail(f"命令 {command} 在 JSON 模式下输出非有效 JSON")

           # JSON 应包含必需字段
           assert 'success' in output_data
           if output_data['success']:
               assert 'data' in output_data
           else:
               assert 'error' in output_data
   ```

   def test_error_responses_json_format(runner):

       """验证参数优先级: --json-output > --quiet > --verbose"""
       with patch('gallery_dl_auo.api.pixiv_client.PixivClient'):
           # --quiet --verbose → 静默模式
           result = runner.invoke(cli, ['--quiet', '--verbose', 'version'])
           assert result.stderr == ""

           # --json-output --quiet → JSON 模式
           result = runner.invoke(cli, ['--json-output', '--quiet', 'version'])

           assert 'success' in output

           # --json-output --verbose → JSON 模式(verbose 被忽略)
           result = runner.invoke(cli, ['--json-output', '--verbose', 'version'])
