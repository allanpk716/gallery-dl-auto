def setup_logging(log_level: str = "INFO", verbose: bool = False) -> Console:
    """配置日志系统

    Args:
        log_level: 日志级别 (DEBUG, INFO, WARNING, ERROR, CRITICAL)
        verbose: 是否启用详细模式(控制台输出)

    Returns:
        Console: Rich Console 实例

    日志策略:
    - 详细模式(verbose=True):控制台 INFO+,文件 DEBUG+
    - 静默模式(verbose=False):控制台无输出,文件 DEBUG+
    """
    console = Console(stderr=True)
    logger = logging.getLogger("gallery_dl_auo")
    logger.setLevel(logging.DEBUG)  # 始终捕获所有级别

    # 清除现有 handlers,避免重复日志
    logger.handlers.clear()

    # 详细模式:控制台输出
    if verbose:
        console_handler = RichHandler(
            console=console,
            show_time=True,
            show_level=True,
            show_path=False,
            markup=True,
            rich_tracebacks=True,
        )
        console_handler.setLevel(logging.INFO)
        logger.addHandler(console_handler)

    # 文件日志始终启用
    from gallery_dl_auo.config.paths import get_log_file_path
    file_handler = StructuredFileHandler(get_log_file_path())
    file_handler.setLevel(logging.DEBUG)
    logger.addHandler(file_handler)

    logger.propagate = False
    return console
```

import requests
from gallery_dl_auo.models.error_response import StructuredError
from gallery_dl_auo.utils.error_codes import ErrorCode

def download_file(url: str, filepath: Path, illust_id: int) -> dict | StructuredError:
    """下载文件,检测 429 速率限制错误

    Args:
        url: 图片 URL
        filepath: 保存路径
        illust_id: 作品 ID

    Returns:
        成功: {"success": True, "filepath": Path, "size": int}
        失败: StructuredError (429 错误或其他错误)
    """
    try:
        response = requests.get(url, timeout=30)

        # 检测 429 速率限制错误
        if response.status_code == 429:
            logger.error(f"Rate limit exceeded (HTTP 429) for illust {illust_id}")
            return StructuredError(
                error_code=ErrorCode.RATE_LIMIT_EXCEEDED,
                error_type="RateLimitError",
                message="Pixiv API 速率限制触发 (HTTP 429)",
                suggestion=(
                    "建议增加延迟参数以避免触发反爬虫机制:\n"
                    "  --image-delay 5.0  (当前: 2.5s)\n"
                    "  --batch-delay 4.0  (当前: 2.0s)"
                ),
                severity="error",
                illust_id=illust_id,
            )

        response.raise_for_status()

        # 原子文件操作:下载到 .tmp 文件
        temp_filepath = filepath.with_suffix(filepath.suffix + ".tmp")
        with open(temp_filepath, "wb") as f:
            f.write(response.content)

        # 原子重命名(Windows 兼容)
        if filepath.exists():
            filepath.unlink()
        temp_filepath.rename(filepath)

        return {
            "success": True,
            "filepath": filepath,
            "size": len(response.content)
        }

    except requests.RequestException as e:
        logger.error(f"Download failed for {url}: {e}")
        return StructuredError(
            error_code=ErrorCode.DOWNLOAD_NETWORK_ERROR,
            error_type="NetworkError",
            message=f"网络错误: {e}",
            suggestion="检查网络连接或稍后重试",
            severity="error",
            illust_id=illust_id,
            original_error=str(e),
        )
```
