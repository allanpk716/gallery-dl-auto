1. **RankingDownloader 类**:
```python
from pathlib import Path
from typing import Optional
import datetime

from gallery_dl_auo.api.pixiv_client import PixivClient, PixivAPIError
from gallery_dl_auo.download.file_downloader import download_file
from gallery_dl_auo.download.rate_limiter import rate_limit_delay
from gallery_dl_auo.utils.filename_sanitizer import sanitize_filename

class RankingDownloader:
    def __init__(self, client: PixivClient, output_dir: Path):
        """
        Args:
            client: Pixiv API 客户端
            output_dir: 输出目录 (默认 ./pixiv-downloads/)
        """
        self.client = client
        self.output_dir = output_dir

    def download_ranking(
        self,
        mode: str = "day",
        date: Optional[str] = None,
        base_delay: float = 2.5,
        jitter: float = 1.0
    ) -> dict:
        """
        下载指定排行榜的所有图片。

        Args:
            mode: 排行榜类型 (day, week, month) - Phase 3 仅实现 day
            date: 日期字符串 YYYY-MM-DD (默认今天)
            base_delay: 基础延迟秒数
            jitter: 随机抖动范围

        Returns:
            dict: {
                'total': int,
                'success': [
                    {
                        'illust_id': int,
                        'title': str,
                        'author': str,
                        'filepath': str
                    }
                ],
                'failed': [
                    {
                        'illust_id': int,
                        'title': str,
                        'error': str
                    }
                ]
            }
        """
        # 1. 获取排行榜数据
        if date is None:
            date = datetime.date.today().strftime("%Y-%m-%d")

        try:
            ranking_data = self.client.get_ranking(mode=mode, date=date)
        except PixivAPIError as e:
            return {
                'total': 0,
                'success': [],
                'failed': [],
                'error': f"API 错误: {e}"
            }

        # 2. 创建输出目录: daily-YYYY-MM-DD/
        ranking_dir = self.output_dir / f"{mode}-{date}"
        ranking_dir.mkdir(parents=True, exist_ok=True)

        # 3. 遍历排行榜,下载每张图片
        results = {
            'total': len(ranking_data),
            'success': [],
            'failed': []
        }

        for illust in ranking_data:
            # 构建文件名: 作品ID_标题.jpg
            filename = sanitize_filename(f"{illust['id']}_{illust['title']}.jpg")
            filepath = ranking_dir / filename

            # 下载图片
            result = download_file(illust['image_url'], filepath)

            if result['success']:
                results['success'].append({
                    'illust_id': illust['id'],
                    'title': illust['title'],
                    'author': illust['author'],
                    'filepath': result['filepath']
                })
            else:
                results['failed'].append({
                    'illust_id': illust['id'],
                    'title': illust['title'],
                    'error': result['error']
                })

            # 速率控制
            rate_limit_delay(base_delay, jitter)

        return results
```

1. **download 命令实现**:
```python
import json
import datetime
import click
from pathlib import Path

from gallery_dl_auo.api.pixiv_client import PixivClient
from gallery_dl_auo.auth.token_storage import get_default_token_storage
from gallery_dl_auo.download.ranking_downloader import RankingDownloader

@click.command()
@click.option(
    "--date",
    default=None,
    help="Download ranking for specific date (YYYY-MM-DD, default: today)"
)
@click.option(
    "--output",
    "-o",
    type=click.Path(),
    default="./pixiv-downloads",
    help="Output directory (default: ./pixiv-downloads)"
)
@click.option(
    "--mode",
    default="day",
    help="Ranking mode (day/week/month, default: day)"
)
def download(date: str | None, output: str, mode: str) -> None:
    """
    Download Pixiv ranking images.

    Downloads images from the specified ranking to the output directory.
    Outputs JSON-formatted results for third-party integration.
    """
    # 1. 加载 token
    storage = get_default_token_storage()
    token_data = storage.load_token()

    if not token_data or not token_data.get("refresh_token"):
        result = {
            "success": False,
            "error": "No token found. Run 'pixiv-downloader login' first."
        }
        print(json.dumps(result, ensure_ascii=False))
        raise SystemExit(1)

    # 2. 初始化 API 客户端
    try:
        client = PixivClient(refresh_token=token_data["refresh_token"])
    except Exception as e:
        result = {
            "success": False,
            "error": f"Authentication failed: {e}"
        }
        print(json.dumps(result, ensure_ascii=False))
        raise SystemExit(1)

    # 3. 执行下载
    output_dir = Path(output)
    downloader = RankingDownloader(client, output_dir)

    results = downloader.download_ranking(
        mode=mode,
        date=date
    )

    # 4. 输出 JSON 结果 (使用 print 而非 console.print,避免 ANSI 转义)
    output_data = {
        "success": True,
        "total": results["total"],
        "success_count": len(results["success"]),
        "failed_count": len(results["failed"]),
        "output_dir": str(output_dir),
        "date": date or datetime.date.today().strftime("%Y-%m-%d"),
        "mode": mode,
        "success_list": results["success"],
        "failed_list": results["failed"]
    }

    print(json.dumps(output_data, ensure_ascii=False, indent=2))

    # 5. 如果有失败,返回退出码 1 (让第三方程序决定重试策略)
    if results["failed"]:
        raise SystemExit(1)
```
