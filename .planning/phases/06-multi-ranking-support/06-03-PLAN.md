```python
from pydantic import BaseModel, Field

class DownloadConfig(BaseModel):
    """下载配置模型"""
    batch_size: int = Field(
        default=30,
        ge=1,
        le=100,
        description="每批次下载的作品数量"
    )
    batch_delay: float = Field(
        default=2.0,
        ge=0.0,
        description="批次间隔秒数"
    )
    concurrency: int = Field(
        default=1,
        ge=1,
        le=10,
        description="并发下载数(Phase 6 仅支持 1)"
    )
    image_delay: float = Field(
        default=2.5,
        ge=0.0,
        description="单张图片间隔秒数"
    )
    max_retries: int = Field(
        default=3,
        ge=1,
        le=10,
        description="单张图片最大重试次数"
    )
    retry_delay: float = Field(
        default=5.0,
        ge=0.0,
        description="重试间隔秒数"
    )

    model_config = {
        "validate_default": True,
        "extra": "forbid"
    }
```

```python
import click
from omegaconf import DictConfig
from gallery_dl_auo.config.download_config import DownloadConfig

@download.command()
@click.option(
    '--type',
    callback=validate_type_param,
    required=True,
    help='排行榜类型 (daily, weekly, monthly, ...)'
)
@click.option(
    '--date',
    callback=validate_date_param,
    default=None,
    help='历史排行榜日期 (YYYY-MM-DD)'
)
@click.option('--output', '-o', default='./pixiv-downloads', help='输出目录')
@click.option('--template', '-t', default=None, help='路径模板')
@click.pass_obj
def download(config: DictConfig, type: str, date: str | None, output: str, template: str | None):
    """下载指定排行榜的图片

    下载参数 (batch_size, image_delay 等) 从配置文件读取。
    配置文件路径: config/download.yaml
    """
    from pathlib import Path
    from gallery_dl_auo.api.pixiv_client import PixivClient
    from gallery_dl_auo.download.ranking_downloader import RankingDownloader

    # 加载下载配置
    download_config = DownloadConfig(**config.get('download', {}))

    # 初始化客户端和下载器
    client = PixivClient(refresh_token="...")  # 从配置或环境变量读取
    output_dir = Path(output)

    downloader = RankingDownloader(
        client=client,
        output_dir=output_dir,
        config=download_config
    )

    # 执行下载
    result = downloader.download_ranking(
        mode=type,  # type 已经通过验证器转换为 mode
        date=date,
        path_template=template
    )

    # 输出结果 (JSON 格式)
    import json
    print(json.dumps(result, ensure_ascii=False, indent=2))

    # 返回退出码
    if result.get('error') or result['failed']:
        raise SystemExit(1)
```
