创建 TokenStorage 类,实现加密 token 存储:

1. **密钥生成逻辑** `_derive_machine_key() -> bytes`:
   ```python
   import hashlib
   import base64
   import socket
   import os
   from pathlib import Path

   def _derive_machine_key() -> bytes:
       """基于机器唯一信息派生 Fernet 密钥"""
       # 收集机器标识
       hostname = socket.gethostname()
       username = os.getenv('USERNAME') or os.getenv('USER') or 'unknown'
       machine_id = hostname  # 或使用 py-machineid 库

       # 组合并哈希生成 32 字节密钥
       seed = f"{hostname}:{username}:{machine_id}".encode('utf-8')
       key_bytes = hashlib.sha256(seed).digest()

       # Fernet 要求 base64url 编码的 32 字节密钥
       return base64.urlsafe_b64encode(key_bytes)
   ```

2. **TokenStorage 类**:
   ```python
   from cryptography.fernet import Fernet
   import json
   from pathlib import Path
   import platform
   import os

   class TokenStorage:
       def __init__(self, storage_path: Path):
           """初始化 Token 存储"""
           self.storage_path = Path(storage_path)
           self.key = _derive_machine_key()
           self.fernet = Fernet(self.key)

       def save_token(self, refresh_token: str, access_token: str = None):
           """加密并保存 token 到文件"""
           # 构造数据字典
           data = {"refresh_token": refresh_token}
           if access_token:
               data["access_token"] = access_token

           # 加密
           encrypted = self.fernet.encrypt(json.dumps(data).encode('utf-8'))

           # 确保目录存在
           self.storage_path.parent.mkdir(parents=True, exist_ok=True)

           # 写入文件
           self.storage_path.write_bytes(encrypted)

           # 设置文件权限 (Unix: 600, Windows: 跳过或使用 ctypes)
           if platform.system() != 'Windows':
               os.chmod(self.storage_path, 0o600)
           else:
               # Windows: 用户目录权限已足够
               pass

       def load_token(self) -> dict | None:
           """解密并加载 token"""
           if not self.storage_path.exists():
               return None

           try:
               encrypted = self.storage_path.read_bytes()
               decrypted = self.fernet.decrypt(encrypted)
               return json.loads(decrypted)
           except Exception as e:
               # 解密失败 (密钥变化、文件损坏等)
               logger.error(f"Failed to decrypt token: {e}")
               return None

       def delete_token(self):
           """删除 token 文件"""
           if self.storage_path.exists():
               self.storage_path.unlink()
   ```

3. **错误处理**:
