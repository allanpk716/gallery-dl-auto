   ```python
   import click
   from rich.console import Console
   from gallery_dl_auo.auth.pixiv_auth import PixivOAuth
   from gallery_dl_auo.auth.token_storage import get_default_token_storage
   from gallery_dl_auo import logger

   @click.command()
   @click.option('--force', is_flag=True, help='Force re-login even if token exists')
   def login(force: bool):
       """Login to Pixiv and save refresh token

       This command will:
       1. Open browser for Pixiv login
       2. Automatically capture refresh token
       3. Encrypt and save token to ~/.gallery-dl-auto/credentials.enc
       """
       console = Console()
       storage = get_default_token_storage()

       # 检查现有 token
       if not force:
           existing = storage.load_token()
           if existing and existing.get('refresh_token'):
               console.print("[yellow]Token already exists. Use --force to re-login.[/yellow]")
               console.print("[dim]Current token status: Use 'pixiv-downloader status' to check[/dim]")
               return

       # 执行登录流程
       try:
           console.print("[bold cyan]Starting Pixiv login...[/bold cyan]")
           console.print("[dim]A browser window will open for you to login.[/dim]")
           console.print("[dim]Please complete the login process in the browser.[/dim]")
           console.print("[dim]The program will automatically detect when login is successful.[/dim]")
           console.print()

           oauth = PixivOAuth()
           tokens = oauth.login()

           # 保存 token
           storage.save_token(
               refresh_token=tokens['refresh_token'],
               access_token=tokens.get('access_token')
           )

           console.print()
           console.print("[bold green]✓ Login successful![/bold green]")
           console.print(f"[dim]Token saved to: {storage.storage_path}[/dim]")
           console.print("[dim]Token is encrypted and secure.[/dim]")

       except Exception as e:
           logger.error(f"Login failed: {e}")
           console.print(f"[bold red]✗ Login failed:[/bold red] {e}")
           console.print("[dim]Please try again. If the problem persists, check your network connection.[/dim]")
           raise click.Abort()
   ```

   ```python
   import click
   from rich.console import Console
   from gallery_dl_auo.auth.token_storage import get_default_token_storage

   @click.command()
   def logout():
       """Logout from Pixiv and delete saved token

       This will remove the encrypted token file from ~/.gallery-dl-auto/credentials.enc
       """
       console = Console()
       storage = get_default_token_storage()

       # 检查 token 是否存在
       if not storage.storage_path.exists():
           console.print("[yellow]No token found. Already logged out.[/yellow]")
           return

       # 确认删除
       if not click.confirm('Are you sure you want to logout?'):
           console.print("[dim]Logout cancelled.[/dim]")
           return

       # 删除 token
       try:
           storage.delete_token()
           console.print("[bold green]✓ Logged out successfully![/bold green]")
           console.print("[dim]Token file has been removed.[/dim]")
           console.print("[dim]Run 'pixiv-downloader login' to login again.[/dim]")
       except Exception as e:
           console.print(f"[bold red]✗ Logout failed:[/bold red] {e}")
           raise click.Abort()
   ```

1. **在 pixiv_auth.py 中添加 token 验证方法**:
   ```python
   @staticmethod
   def validate_refresh_token(refresh_token: str) -> dict:
       """验证 refresh_token 是否有效,返回 token 信息

       Returns:
           dict: {
               'valid': bool,
               'access_token': str | None,
               'refresh_token': str | None,
               'expires_in': int | None,
               'error': str | None
           }
       """
       try:
           response = requests.post(
               TOKEN_URL,
               data={
                   "client_id": CLIENT_ID,
                   "client_secret": CLIENT_SECRET,
                   "grant_type": "refresh_token",
                   "refresh_token": refresh_token,
               },
               headers={"User-Agent": "PixivAndroidApp/5.0.64 (Android 6.0)"},
           )

           if response.status_code == 200:
               data = response.json()
               return {
                   'valid': True,
                   'access_token': data['access_token'],
                   'refresh_token': data['refresh_token'],
                   'expires_in': data.get('expires_in', 3600),
                   'error': None
               }
           else:
               error_data = response.json()
               return {
                   'valid': False,
                   'access_token': None,
                   'refresh_token': None,
                   'expires_in': None,
                   'error': error_data.get('error', 'Unknown error')
               }
       except Exception as e:
           return {
               'valid': False,
               'access_token': None,
               'refresh_token': None,
               'expires_in': None,
               'error': str(e)
           }
   ```

2. **Status 命令实现**:
   ```python
   import click
   from rich.console import Console
   from rich.table import Table
   from gallery_dl_auo.auth.token_storage import get_default_token_storage
   from gallery_dl_auo.auth.pixiv_auth import PixivOAuth

   @click.command()
   @click.option('--verbose', '-v', is_flag=True, help='Show detailed token information')
   def status(verbose: bool):
       """Check Pixiv token status

       Shows whether a valid token exists and can be refreshed
       """
       console = Console()
       storage = get_default_token_storage()

       # 检查 token 文件是否存在
       if not storage.storage_path.exists():
           console.print("[yellow]No token found.[/yellow]")
           console.print("[dim]Run 'pixiv-downloader login' to login.[/dim]")
           return

       # 加载 token
       token_data = storage.load_token()
       if not token_data:
           console.print("[red]✗ Token file exists but cannot be decrypted.[/red]")
           console.print("[dim]This may happen if machine info changed or file is corrupted.[/dim]")
           console.print("[dim]Run 'pixiv-downloader login --force' to re-login.[/dim]")
           return

       refresh_token = token_data.get('refresh_token')
       if not refresh_token:
           console.print("[red]✗ Invalid token data (missing refresh_token).[/red]")
           return

       # 验证 token
       console.print("[dim]Validating token...[/dim]")
       result = PixivOAuth.validate_refresh_token(refresh_token)

       # 显示结果
       table = Table(title="Token Status")
       table.add_column("Property", style="cyan")
       table.add_column("Value", style="green")

       if result['valid']:
           table.add_row("Status", "[green]✓ Valid[/green]")
           table.add_row("Token File", str(storage.storage_path))

           if verbose:
               # 显示部分 token (隐私保护)
               masked_token = refresh_token[:10] + "..." + refresh_token[-10:]
               table.add_row("Refresh Token", masked_token)

               if result.get('expires_in'):
                   import datetime
                   expiry = datetime.datetime.now() + datetime.timedelta(seconds=result['expires_in'])
                   table.add_row("Expires", expiry.strftime('%Y-%m-%d %H:%M:%S'))

           # 如果 token 有效,更新存储 (自动刷新)
           if result['refresh_token']:
               storage.save_token(
                   refresh_token=result['refresh_token'],
                   access_token=result.get('access_token')
               )
               console.print("[dim]Token refreshed and saved.[/dim]")

       else:
           table.add_row("Status", "[red]✗ Invalid[/red]")
           table.add_row("Error", result.get('error', 'Unknown'))
           table.add_row("Suggestion", "Run 'pixiv-downloader login --force' to re-login")

       console.print(table)
   ```
