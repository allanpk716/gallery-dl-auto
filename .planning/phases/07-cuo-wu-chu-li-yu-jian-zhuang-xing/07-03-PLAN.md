## Objective

实现统一的 JSON 错误响应格式,所有错误(网络、权限、API、文件系统)通过结构化 JSON 返回给第三方调用者。错误响应包含:错误码、错误类型、友好消息、建议操作、作品 ID、原始异常。支持批量下载时部分成功场景,返回成功和失败列表。

   ```python
   from pydantic import BaseModel, Field
   from typing import Optional
   from enum import Enum
   from datetime import datetime

   class ErrorSeverity(str, Enum):
       WARNING = "warning"   # 跳过单个项目,继续执行
       ERROR = "error"       # 操作失败,但程序继续
       CRITICAL = "critical" # 程序终止

   class StructuredError(BaseModel):
       """结构化错误模型"""
       error_code: str = Field(..., description="错误码,如 DOWNLOAD_FAILED")
       error_type: str = Field(..., description="错误类型,如 NetworkError")
       message: str = Field(..., description="用户友好的错误消息")
       suggestion: str = Field(..., description="建议的操作步骤")
       severity: ErrorSeverity = Field(..., description="错误严重性")
       illust_id: Optional[int] = Field(None, description="相关作品 ID")
       original_error: Optional[str] = Field(None, description="原始异常信息")
       timestamp: str = Field(
           default_factory=lambda: datetime.now().isoformat(),
           description="错误发生时间"
       )

       class Config:
           json_encoders = {
               ErrorSeverity: lambda v: v.value
           }

   class BatchDownloadResult(BaseModel):
       """批量下载结果模型"""
       success: bool = Field(..., description="整体是否成功")
       total: int = Field(..., description="总作品数")
       downloaded: int = Field(..., description="成功下载数")
       failed: int = Field(..., description="失败数")
       skipped: int = Field(..., description="跳过数(已下载)")
       success_list: list[int] = Field(default_factory=list, description="成功作品 ID")
       failed_errors: list[StructuredError] = Field(
           default_factory=list,
           description="失败错误列表"
       )
       output_dir: str = Field(..., description="输出目录")

       class Config:
           json_encoders = {
               datetime: lambda v: v.isoformat()
           }
   ```

3. 在 `models/__init__.py` 中导出新模型

<action>
1. 修改 `download_file()` 函数签名:
   - 返回类型从 `dict` 改为 `StructuredError | dict`(成功返回 dict,失败返回 StructuredError)

2. 重构异常处理:
   ```python
   def download_file(url: str, output_path: Path, illust_id: int) -> StructuredError | dict:
       """下载文件,返回结构化错误或成功字典"""
       try:
           # ... 下载逻辑 ...

           # 成功
           return {
               "success": True,
               "path": str(output_path),
               "size": file_size
           }

       except PermissionError as e:
           return StructuredError(
               error_code=ErrorCode.DOWNLOAD_PERMISSION_DENIED,
               error_type="PermissionError",
               message=f"无法写入文件: {output_path.name}",
               suggestion="检查目录权限或以管理员身份运行",
               severity=ErrorSeverity.ERROR,
               illust_id=illust_id,
               original_error=str(e)
           )

       except requests.Timeout as e:
           return StructuredError(
               error_code=ErrorCode.DOWNLOAD_TIMEOUT,
               error_type="TimeoutError",
               message=f"下载超时:作品 {illust_id}",
               suggestion="检查网络连接或稍后重试",
               severity=ErrorSeverity.WARNING,
               illust_id=illust_id,
               original_error=str(e)
           )

       except requests.RequestException as e:
           return StructuredError(
               error_code=ErrorCode.DOWNLOAD_FAILED,
               error_type="NetworkError",
               message=f"下载失败:作品 {illust_id}",
               suggestion="检查网络连接,错误将持续重试 3 次",
               severity=ErrorSeverity.WARNING,
               illust_id=illust_id,
               original_error=str(e)
           )
   ```

3. 保留现有的 `@retry_on_file_error` 装饰器,仅在重试耗尽后返回错误

   ```python
   def download_ranking(...) -> BatchDownloadResult:
       errors = []
       success_list = []
       skipped = 0

       for illust in ranking.illusts:
           if tracker.is_downloaded(illust.id):
               skipped += 1
               continue

           result = download_file(...)
           if isinstance(result, StructuredError):
               errors.append(result)
           else:
               success_list.append(illust.id)
               tracker.record_download(...)

       return BatchDownloadResult(
           success=len(errors) == 0,
           total=len(ranking.illusts),
           downloaded=len(success_list),
           failed=len(errors),
           skipped=skipped,
           success_list=success_list,
           failed_errors=errors,
           output_dir=str(output_dir)
       )
   ```
